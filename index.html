async function startQuiz() { const name = document.getElementById('userName').value.trim(); const mobile = document.getElementById('userMobile').value.trim(); const dist = document.getElementById('userDistrict').value; if(!name || mobile.length !== 10 || !dist) return alert("Details invalid."); document.getElementById('display-subject').innerText = "CONNECTING..."; const qId = new URLSearchParams(window.location.search).get('id') || "CIVICS_CLASS8_T2"; let attempts = 0; let success = false; while (attempts < 3 && !success) { try { attempts++; const qSnap = await window.getDocs(window.query(window.collection(window.db, "quizzes", qId, "questions"), window.orderBy("qno"))); questions = qSnap.docs.map(doc => doc.data()); if (questions.length > 0) success = true; else await new Promise(r => setTimeout(r, 1500)); } catch (e) { await new Promise(r => setTimeout(r, 1500)); } } document.getElementById('display-subject').innerText = qId.replace(/_/g, ' '); if (!success || questions.length === 0) { return alert("QUIZ NOT FOUND:\nCheck that your link parameters match your Sheet tab casing perfectly."); } // --- FIX: MULTI-USER MEMORY OVERLAP PREVENTION --- const previousSession = localStorage.getItem('tnSession'); let shouldRestore = false; if (previousSession) { const parsedSession = JSON.parse(previousSession); // Only restore progress if it is the EXACT same student continuing the same quiz if (parsedSession.name === name && parsedSession.mobile === mobile && parsedSession.qId === qId) { shouldRestore = true; } } if (shouldRestore) { const savedAnswers = localStorage.getItem('tnAnswers'); answers = savedAnswers && JSON.parse(savedAnswers).length === questions.length ? JSON.parse(savedAnswers) : new Array(questions.length).fill(null); } else { // Fresh session for a new user: Wipe old data completely out of browser memory localStorage.removeItem('tnAnswers'); answers = new Array(questions.length).fill(null); } // ------------------------------------------------- localStorage.setItem('tnSession', JSON.stringify({name, mobile, dist, qId})); skipped = new Array(questions.length).fill(false); document.getElementById('section-reg').classList.add('hidden'); document.getElementById('section-quiz').classList.remove('hidden'); showQuestion(); }
